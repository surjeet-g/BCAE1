              errorMsg = "Camara unavailable";

              break;
            case "permission":
              errorMsg = "Permission not grant";

              break;
            case "others":
              errorMsg = "Technical error";

              break;

            default:
              errorMsg = "Technical error";
              break;
          }
          Toast.show({
            type: "bctError",
            text1: errorMsg,
          });
        } else {
          // const source = { uri: response.uri };

            setFileAttachments([
              ...fileAttachments,
              {
                fileType: response?.assets[0]?.type,
